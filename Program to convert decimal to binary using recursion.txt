int convert(int n, int rem[50], int i){
    int d, j;
    while(n > 0){
        d = n % 2;
        rem[i] = d;
        i++;
        n = n / 2;
        convert(n, rem, i);
    }
    for(j = 0; j <= i; j++){
        printf("%d", rem[i-j]);
    }
    return(rem);
}
